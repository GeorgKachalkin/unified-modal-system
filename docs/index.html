📦 Подключение в проекте

<!-- 1. Конфиг (до um-core.js) -->
<script>
  window.UM_CONFIG = { jquery_ui: false };
  var lang = { close: 'Закрыть', save: 'Сохранить', cancel: 'Отмена', insert: 'Вставить' };
</script>

<!-- 2. Стили и скрипт -->
<link rel="stylesheet" href="../css/um-core.css">
<script src="../js/um-core.js"></script>

<!-- 3. Использование -->
<script>
  UM_Alert('success', 'Готово!');
  UM_ShowModal({ title: 'Привет', descr: 'Текст окна' });
</script>
